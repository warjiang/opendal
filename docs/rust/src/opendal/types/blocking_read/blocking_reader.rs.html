18use std::collections::Bound;
19use std::ops::Range;
20use std::ops::RangeBounds;
21use std::sync::Arc;
22
23use bytes::BufMut;
24
25use crate::raw::*;
26use crate::*;
27
28/// BlockingReader is designed to read data from given path in an blocking
29/// manner.
30#[derive(Clone)]
31pub struct BlockingReader {
32    ctx: Arc<ReadContext>,
33
34    /// Total size of the reader.
35    size: Arc<AtomicContentLength>,
36}
37
38impl BlockingReader {
39    /// Create a new blocking reader.
40    ///
41    /// Create will use internal information to decide the most suitable
42    /// implementation for users.
43    ///
44    /// We don't want to expose those details to users so keep this function
45    /// in crate only.
46    pub(crate) fn new(ctx: ReadContext) -> Self {
47        BlockingReader {
48            ctx: Arc::new(ctx),
49            size: Arc::new(AtomicContentLength::new()),
50        }
51    }
52
53    /// Parse users input range bounds into valid `Range<u64>`.
54    ///
55    /// To avoid duplicated stat call, we will cache the size of the reader.
56    fn parse_range(&self, range: impl RangeBounds<u64>) -> Result<Range<u64>> {
57        let start = match range.start_bound() {
58            Bound::Included(v) => *v,
59            Bound::Excluded(v) => v + 1,
60            Bound::Unbounded => 0,
61        };
62
63        let end = match range.end_bound() {
64            Bound::Included(v) => v + 1,
65            Bound::Excluded(v) => *v,
66            Bound::Unbounded => match self.size.load() {
67                Some(v) => v,
68                None => {
69                    let size = self
70                        .ctx
71                        .accessor()
72                        .blocking_stat(self.ctx.path(), OpStat::new())?
73                        .into_metadata()
74                        .content_length();
75                    self.size.store(size);
76                    size
77                }
78            },
79        };
80
81        Ok(start..end)
82    }
83
84    /// Read give range from reader into [`Buffer`].
85    ///
86    /// This operation is zero-copy, which means it keeps the [`bytes::Bytes`] returned by underlying
87    /// storage services without any extra copy or intensive memory allocations.
88    ///
89    /// # Notes
90    ///
91    /// - Buffer length smaller than range means we have reached the end of file.
92    pub fn read(&self, range: impl RangeBounds<u64>) -> Result<Buffer> {
93        let mut bufs = vec![];
94        for buf in self.clone().into_iterator(range)? {
95            bufs.push(buf?)
96        }
97        Ok(bufs.into_iter().flatten().collect())
98    }
99
100    ///
101    /// This operation will copy and write bytes into given [`BufMut`]. Allocation happens while
102    /// [`BufMut`] doesn't have enough space.
103    ///
104    /// # Notes
105    ///
106    /// - Returning length smaller than range means we have reached the end of file.
107    pub fn read_into(&self, buf: &mut impl BufMut, range: impl RangeBounds<u64>) -> Result<usize> {
108        let mut iter = self.clone().into_iterator(range)?;
109
110        let mut read = 0;
111        loop {
112            let Some(bs) = iter.next().transpose()? else {
113                return Ok(read);
114            };
115            read += bs.len();
116            buf.put(bs);
117        }
118    }
119
120    /// Create a buffer iterator to read specific range from given reader.
121    fn into_iterator(self, range: impl RangeBounds<u64>) -> Result<BufferIterator> {
122        let range = self.parse_range(range)?;
123        Ok(BufferIterator::new(self.ctx, range))
124    }
125
126    /// Convert reader into [`StdReader`] which implements [`futures::AsyncRead`],
127    /// [`futures::AsyncSeek`] and [`futures::AsyncBufRead`].
128    #[inline]
129    pub fn into_std_read(self, range: impl RangeBounds<u64>) -> Result<StdReader> {
130        let range = self.parse_range(range)?;
131        Ok(StdReader::new(self.ctx, range))
132    }
133
134    /// Convert reader into [`StdBytesIterator`] which implements [`Iterator`].
135    #[inline]
136    pub fn into_bytes_iterator(self, range: impl RangeBounds<u64>) -> Result<StdBytesIterator> {
137        let range = self.parse_range(range)?;
138        Ok(StdBytesIterator::new(self.ctx, range))
139    }
140}
