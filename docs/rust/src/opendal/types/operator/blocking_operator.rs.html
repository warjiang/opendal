23/// BlockingOperator is the entry for all public blocking APIs.
24///
25/// Read [`concepts`][docs::concepts] for know more about [`Operator`].
26///
27/// # Examples
28///
29/// ## Init backends
30///
31/// Read more backend init examples in [`services`]
32///
33/// ```rust,no_run
34/// # use anyhow::Result;
35/// use opendal::services::Fs;
36/// use opendal::BlockingOperator;
37/// use opendal::Operator;
38///
39/// fn main() -> Result<()> {
40///     // Create fs backend builder.
41///     let builder = Fs::default().root("/tmp");
42///
43///     // Build an `BlockingOperator` to start operating the storage.
44///     let _: BlockingOperator = Operator::new(builder)?.finish().blocking();
45///
46///     Ok(())
47/// }
48/// ```
49///
50/// ## Init backends with blocking layer
51///
52/// Some services like s3, gcs doesn't have native blocking supports, we can use [`layers::BlockingLayer`]
53/// to wrap the async operator to make it blocking.
54#[cfg_attr(feature = "layers-blocking", doc = "```rust")]
55#[cfg_attr(not(feature = "layers-blocking"), doc = "```ignore")]
56/// # use anyhow::Result;
57/// use opendal::layers::BlockingLayer;
58/// use opendal::services::S3;
59/// use opendal::BlockingOperator;
60/// use opendal::Operator;
61///
62/// async fn test() -> Result<()> {
63///     // Create fs backend builder.
64///     let mut builder = S3::default().bucket("test").region("us-east-1");
65///
66///     // Build an `BlockingOperator` with blocking layer to start operating the storage.
67///     let _: BlockingOperator = Operator::new(builder)?
68///         .layer(BlockingLayer::create()?)
69///         .finish()
70///         .blocking();
71///
72///     Ok(())
73/// }
74/// ```
75#[derive(Clone, Debug)]
76pub struct BlockingOperator {
77    accessor: Accessor,
78}

128    /// Get given path's metadata.
129    ///
130    /// # Behavior
131    ///
132    /// ## Services that support `create_dir`
133    ///
134    /// `test` and `test/` may vary in some services such as S3. However, on a local file system,
135    /// they're identical. Therefore, the behavior of `stat("test")` and `stat("test/")` might differ
136    /// in certain edge cases. Always use `stat("test/")` when you need to access a directory if possible.
137    ///
138    /// Here are the behavior list:
139    ///
140    /// | Case                   | Path            | Result                                     |
141    /// |------------------------|-----------------|--------------------------------------------|
142    /// | stat existing dir      | `abc/`          | Metadata with dir mode                     |
143    /// | stat existing file     | `abc/def_file`  | Metadata with file mode                    |
144    /// | stat dir without `/`   | `abc/def_dir`   | Error `NotFound` or metadata with dir mode |
145    /// | stat file with `/`     | `abc/def_file/` | Error `NotFound`                           |
146    /// | stat not existing path | `xyz`           | Error `NotFound`                           |
147    ///
148    /// Refer to [RFC: List Prefix][crate::docs::rfcs::rfc_3243_list_prefix] for more details.
149    ///
150    /// ## Services that not support `create_dir`
151    ///
152    /// For services that not support `create_dir`, `stat("test/")` will return `NotFound` even
153    /// when `test/abc` exists since the service won't have the concept of dir. There is nothing
154    /// we can do about this.
155    ///
156    /// # Examples
157    ///
158    /// ## Check if file exists
159    ///
160    /// ```
161    /// # use anyhow::Result;
162    /// # use futures::io;
163    /// # use opendal::BlockingOperator;
164    /// use opendal::ErrorKind;
165    /// #
166    /// # fn test(op: BlockingOperator) -> Result<()> {
167    /// if let Err(e) = op.stat("test") {
168    ///     if e.kind() == ErrorKind::NotFound {
169    ///         println!("file not exist")
170    ///     }
171    /// }
172    /// # Ok(())
173    /// # }
174    /// ```
175    pub fn stat(&self, path: &str) -> Result<Metadata> {
176        self.stat_with(path).call()
177    }

179    /// Get given path's metadata with extra options.
180    ///
181    /// # Behavior
182    ///
183    /// ## Services that support `create_dir`
184    ///
185    /// `test` and `test/` may vary in some services such as S3. However, on a local file system,
186    /// they're identical. Therefore, the behavior of `stat("test")` and `stat("test/")` might differ
187    /// in certain edge cases. Always use `stat("test/")` when you need to access a directory if possible.
188    ///
189    /// Here are the behavior list:
190    ///
191    /// | Case                   | Path            | Result                                     |
192    /// |------------------------|-----------------|--------------------------------------------|
193    /// | stat existing dir      | `abc/`          | Metadata with dir mode                     |
194    /// | stat existing file     | `abc/def_file`  | Metadata with file mode                    |
195    /// | stat dir without `/`   | `abc/def_dir`   | Error `NotFound` or metadata with dir mode |
196    /// | stat file with `/`     | `abc/def_file/` | Error `NotFound`                           |
197    /// | stat not existing path | `xyz`           | Error `NotFound`                           |
198    ///
199    /// Refer to [RFC: List Prefix][crate::docs::rfcs::rfc_3243_list_prefix] for more details.
200    ///
201    /// ## Services that not support `create_dir`
202    ///
203    /// For services that not support `create_dir`, `stat("test/")` will return `NotFound` even
204    /// when `test/abc` exists since the service won't have the concept of dir. There is nothing
205    /// we can do about this.
206    ///
207    /// # Examples
208    ///
209    /// ## Get metadata while `ETag` matches
210    ///
211    /// `stat_with` will
212    ///
213    /// - return `Ok(metadata)` if `ETag` matches
214    /// - return `Err(error)` and `error.kind() == ErrorKind::ConditionNotMatch` if file exists but
215    ///   `ETag` mismatch
216    /// - return `Err(err)` if other errors occur, for example, `NotFound`.
217    ///
218    /// ```
219    /// # use anyhow::Result;
220    /// # use opendal::BlockingOperator;
221    /// use opendal::ErrorKind;
222    /// #
223    /// # fn test(op: BlockingOperator) -> Result<()> {
224    /// if let Err(e) = op.stat_with("test").if_match("<etag>").call() {
225    ///     if e.kind() == ErrorKind::ConditionNotMatch {
226    ///         println!("file exists, but etag mismatch")
227    ///     }
228    ///     if e.kind() == ErrorKind::NotFound {
229    ///         println!("file not exist")
230    ///     }
231    /// }
232    /// # Ok(())
233    /// # }
234    /// ```
235    pub fn stat_with(&self, path: &str) -> FunctionStat {
236        let path = normalize_path(path);
237
238        FunctionStat(OperatorFunction::new(
239            self.inner().clone(),
240            path,
241            OpStat::default(),
242            |inner, path, args| {
243                let rp = inner.blocking_stat(&path, args)?;
244                let meta = rp.into_metadata();
245
246                Ok(meta)
247            },
248        ))
249    }

252    ///
253    /// # Example
254    ///
255    /// ```no_run
256    /// use anyhow::Result;
257    /// use opendal::BlockingOperator;
258    /// fn test(op: BlockingOperator) -> Result<()> {
259    ///     let _ = op.exists("test")?;

264    pub fn exists(&self, path: &str) -> Result<bool> {
265        let r = self.stat(path);
266        match r {
267            Ok(_) => Ok(true),
268            Err(err) => match err.kind() {
269                ErrorKind::NotFound => Ok(false),
270                _ => Err(err),
271            },
272        }

293    /// Create a dir at given path.
294    ///
295    /// # Notes
296    ///
297    /// To indicate that a path is a directory, it is compulsory to include
298    /// a trailing / in the path. Failure to do so may result in
299    /// `NotADirectory` error being returned by OpenDAL.
300    ///
301    /// # Behavior
302    ///
303    /// - Create on existing dir will succeed.
304    /// - Create dir is always recursive, works like `mkdir -p`
305    ///
306    /// # Examples
307    ///
308    /// ```no_run
309    /// # use opendal::Result;
310    /// # use opendal::BlockingOperator;
311    /// # use futures::TryStreamExt;
312    /// # fn test(op: BlockingOperator) -> Result<()> {
313    /// op.create_dir("path/to/dir/")?;
314    /// # Ok(())
315    /// # }
316    /// ```
317    pub fn create_dir(&self, path: &str) -> Result<()> {
318        let path = normalize_path(path);
319
320        if !validate_path(&path, EntryMode::DIR) {
321            return Err(Error::new(
322                ErrorKind::NotADirectory,
323                "the path trying to create should end with `/`",
324            )
325            .with_operation("create_dir")
326            .with_context("service", self.inner().info().scheme())
327            .with_context("path", &path));
328        }
329
330        self.inner()
331            .blocking_create_dir(&path, OpCreateDir::new())?;
332
333        Ok(())
334    }

356    /// Read the whole path into a bytes with extra options.
357    ///
358    /// This function will allocate a new bytes internally. For more precise memory control or
359    /// reading data lazily, please use [`BlockingOperator::reader`]
360    ///
361    /// # Examples
362    ///
363    /// ```no_run
364    /// # use anyhow::Result;
365    /// use opendal::BlockingOperator;
366    /// use opendal::EntryMode;
367    /// # fn test(op: BlockingOperator) -> Result<()> {
368    /// let bs = op.read_with("path/to/file").range(0..10).call()?;
369    /// # Ok(())
370    /// # }
371    /// ```
372    pub fn read_with(&self, path: &str) -> FunctionRead {
373        let path = normalize_path(path);
374
375        FunctionRead(OperatorFunction::new(
376            self.inner().clone(),
377            path,
378            (OpRead::default(), BytesRange::default()),
379            |inner, path, (args, range)| {
380                if !validate_path(&path, EntryMode::FILE) {
381                    return Err(
382                        Error::new(ErrorKind::IsADirectory, "read path is a directory")
383                            .with_operation("BlockingOperator::read_with")
384                            .with_context("service", inner.info().scheme().into_static())
385                            .with_context("path", &path),
386                    );
387                }
388
389                let context = ReadContext::new(inner, path, args, OpReader::default());
390                let r = BlockingReader::new(context);
391                let buf = r.read(range.to_range())?;
392                Ok(buf)
393            },
394        ))
395    }

477    /// Copy a file from `from` to `to`.
478    ///
479    /// # Notes
480    ///
481    /// - `from` and `to` must be a file.
482    /// - `to` will be overwritten if it exists.
483    /// - If `from` and `to` are the same, nothing will happen.
484    /// - `copy` is idempotent. For same `from` and `to` input, the result will be the same.
485    ///
486    /// # Examples
487    ///
488    /// ```
489    /// # use opendal::Result;
490    /// # use opendal::BlockingOperator;
491    ///
492    /// # fn test(op: BlockingOperator) -> Result<()> {
493    /// op.copy("path/to/file", "path/to/file2")?;
494    /// # Ok(())
495    /// # }
496    /// ```
497    pub fn copy(&self, from: &str, to: &str) -> Result<()> {
498        let from = normalize_path(from);
499
500        if !validate_path(&from, EntryMode::FILE) {
501            return Err(
502                Error::new(ErrorKind::IsADirectory, "from path is a directory")
503                    .with_operation("BlockingOperator::copy")
504                    .with_context("service", self.info().scheme())
505                    .with_context("from", from),
506            );
507        }
508
509        let to = normalize_path(to);
510
511        if !validate_path(&to, EntryMode::FILE) {
512            return Err(
513                Error::new(ErrorKind::IsADirectory, "to path is a directory")
514                    .with_operation("BlockingOperator::copy")
515                    .with_context("service", self.info().scheme())
516                    .with_context("to", to),
517            );
518        }
519
520        if from == to {
521            return Err(
522                Error::new(ErrorKind::IsSameFile, "from and to paths are same")
523                    .with_operation("BlockingOperator::copy")
524                    .with_context("service", self.info().scheme())
525                    .with_context("from", from)
526                    .with_context("to", to),
527            );
528        }
529
530        self.inner().blocking_copy(&from, &to, OpCopy::new())?;
531
532        Ok(())
533    }

535    /// Rename a file from `from` to `to`.
536    ///
537    /// # Notes
538    ///
539    /// - `from` and `to` must be a file.
540    /// - `to` will be overwritten if it exists.
541    /// - If `from` and `to` are the same, a `IsSameFile` error will occur.
542    ///
543    /// # Examples
544    ///
545    /// ```
546    /// # use opendal::Result;
547    /// # use opendal::BlockingOperator;
548    ///
549    /// # fn test(op: BlockingOperator) -> Result<()> {
550    /// op.rename("path/to/file", "path/to/file2")?;
551    /// # Ok(())
552    /// # }
553    /// ```
554    pub fn rename(&self, from: &str, to: &str) -> Result<()> {
555        let from = normalize_path(from);
556
557        if !validate_path(&from, EntryMode::FILE) {
558            return Err(
559                Error::new(ErrorKind::IsADirectory, "from path is a directory")
560                    .with_operation("BlockingOperator::move")
561                    .with_context("service", self.info().scheme())
562                    .with_context("from", from),
563            );
564        }
565
566        let to = normalize_path(to);
567
568        if !validate_path(&to, EntryMode::FILE) {
569            return Err(
570                Error::new(ErrorKind::IsADirectory, "to path is a directory")
571                    .with_operation("BlockingOperator::move")
572                    .with_context("service", self.info().scheme())
573                    .with_context("to", to),
574            );
575        }
576
577        if from == to {
578            return Err(
579                Error::new(ErrorKind::IsSameFile, "from and to paths are same")
580                    .with_operation("BlockingOperator::move")
581                    .with_context("service", self.info().scheme())
582                    .with_context("from", from)
583                    .with_context("to", to),
584            );
585        }
586
587        self.inner().blocking_rename(&from, &to, OpRename::new())?;
588
589        Ok(())
590    }

634                let mut w = BlockingWriter::new(context)?;

691            |inner, path, (args, options)| {

696                        Error::new(ErrorKind::IsADirectory, "write path is a directory")
697                            .with_operation("BlockingOperator::writer_with")
698                            .with_context("service", inner.info().scheme().into_static())
699                            .with_context("path", &path),
700                    );
701                }

703                let context = WriteContext::new(inner, path, args, options);

859    /// Remove the path and all nested dirs and files recursively.
860    ///
861    /// # Notes
862    ///
863    /// We don't support batch delete now.
864    ///
865    /// # Examples
866    ///
867    /// ```
868    /// # use anyhow::Result;
869    /// # use futures::io;
870    /// # use opendal::BlockingOperator;
871    /// # fn test(op: BlockingOperator) -> Result<()> {
872    /// op.remove_all("path/to/dir")?;
873    /// # Ok(())
874    /// # }
875    /// ```
876    pub fn remove_all(&self, path: &str) -> Result<()> {
877        match self.stat(path) {
878            Ok(metadata) => {
879                if metadata.mode() != EntryMode::DIR {
880                    self.delete(path)?;
881                    // There may still be objects prefixed with the path in some backend, so we can't return here.
882                }
883            }
884
885            // If dir not found, it may be a prefix in object store like S3,
886            // and we still need to delete objects under the prefix.
887            Err(e) if e.kind() == ErrorKind::NotFound => {}
888
889            Err(e) => return Err(e),
890        };
891
892        let lister = self.lister_with(path).recursive(true).call()?;
893        self.delete_try_iter(lister)?;
894
895        Ok(())
896    }

900    /// # Notes
901    ///
902    /// ## Recursively List
903    ///
904    /// This function only read the children of the given directory. To read
905    /// all entries recursively, use `BlockingOperator::list_with("path").recursive(true)`

934    ///     }
